94       def load_model(self):
95           print("Loading model...")
96           if os.path.exists(self.model_json):
97               with open(self.model_json, "r") as f:
98                   self.model_option = json.load(f)
99           else:
100              with open(self.model_json, "w") as f:
101                  json.dump(self.model_option, f)
102          self.model = KAN(
103              width=[self.len_input, self.model_option["hidden_layer"], 1],
104              grid=self.model_option["grid"],
105              k=self.model_option["k"], auto_save=False, seed=0)
106          if os.path.exists(self.file_model):
107              self.model.load_state_dict(torch.load(self.file_model))
108          if not os.path.exists(self.file_formula):
109              return None
110          else:
111              with open(self.file_formula, encoding="UTF-8", mode="r") as p:
112                  return str(p.read()).strip()

195      def get_state_data(self, state):
196          train_input = [[0.] * 64 for _ in range(12)]
197          for piece in chess.PIECE_TYPES:
198              for square in state.pieces(piece, chess.BLACK):
199                  train_input[piece - 1][square] = -piece
200                  for move in state.pseudo_legal_moves:
201                      if move.from_square == square:
202                          train_input[piece - 1][move.to_square] = -piece
203          for piece in chess.PIECE_TYPES:
204              for square in state.pieces(piece, chess.WHITE):
205                  train_input[piece + 5][square] = piece
206                  for move in state.pseudo_legal_moves:
207                      if move.from_square == square:
208                          train_input[piece + 5][move.to_square] = piece
209          train_input = [j for i in train_input for j in i]
210          if state.has_kingside_castling_rights(state.turn):
211              train_input = [1.] + train_input
212          else:
213              train_input = [0.] + train_input
214          if state.has_kingside_castling_rights(state.turn):
215              train_input = [1.] + train_input
216          else:
217              train_input = [0.] + train_input
218          if state.ep_square is None:
219              train_input = [0.] + train_input
220          else:
221              train_input = [state.ep_square] + train_input
222          train_input = [int(state.turn)] + train_input
223          return train_input[:self.len_input // 2]

225      def get_data(self, fen_generator, get_score, _limit):
226          count = 0
227          count2 = 0
228          dataset = {}
229          train_inputs = []
230          train_labels = []
231          test_inputs = []
232          test_labels = []
233          board = chess.Board()
234          for fen in fen_generator(get_score, _limit):
235              scores = []
236              boards = []
237              try:
238                  board.set_fen(fen)
239                  score = get_score(board)
240                  if score is None:
241                      continue
242              except chess.engine.EngineError:
243                  continue
244              except chess.IllegalMoveError:
245                  continue
246              scores.append(score)
247              boards.append(board.copy())
248              count += 1
249              moves = board.legal_moves
250              for move in moves:
251                  try:
252                      board.push(move)
253                      score = get_score(board)
254                      if score is None:
255                          board.pop()
256                          continue
257                  except chess.engine.EngineError:
258                      break
259                  except chess.IllegalMoveError:
260                      break
261                  scores.append(score)
262                  boards.append(board.copy())
263                  count2 += 1
264                  utils_progress(
265                      f"{str(count).rjust(5, ' ')} | " +
266                      f"{str(count2).rjust(5, ' ')} | " +
267                      f"{str(scores[-1]).rjust(2, ' ')} | {board.fen()}")
268                  board.pop()
269              if count % 2 == 0:
270                  for i in range(1, len(boards)):
271                      test_input = self.get_train(state1=boards[0], state2=boards[i])
272                      test_inputs.append(test_input)
273                      test_labels.append([scores[i] - scores[0]])
274                      test_input = self.get_train(state1=boards[i], state2=boards[0])
275                      test_inputs.append(test_input)
276                      test_labels.append([scores[0] - scores[i]])
277              else:
278                  for i in range(1, len(boards)):
279                      train_input = self.get_train(state1=boards[0], state2=boards[i])
280                      train_inputs.append(train_input)
281                      train_labels.append([scores[i] - scores[0]])
282                      train_input = self.get_train(state1=boards[i], state2=boards[0])
283                      train_inputs.append(train_input)
284                      train_labels.append([scores[0] - scores[i]])
285          print()
286          min_len = min(len(test_inputs), len(train_inputs),
287                        len(test_labels), len(train_labels),
288                        )
289          test_inputs = test_inputs[:min_len]
290          test_labels = test_labels[:min_len]
291          train_inputs = train_inputs[:min_len]
292          train_labels = train_labels[:min_len]
293          dataset['train_input'] = \
294              torch.FloatTensor(train_inputs).type(self.dtype).to(self.device)
295          dataset['train_label'] = \
296              torch.FloatTensor(train_labels).type(self.dtype).to(self.device)
297          dataset['test_input'] = \
298              torch.FloatTensor(test_inputs).type(self.dtype).to(self.device)
299          dataset['test_label'] = \
300              torch.FloatTensor(test_labels).type(self.dtype).to(self.device)
301          return dataset

315      def set_piece(self, state, piece):
316          while True:
317              pos = self.random.choice(list(range(64)))
318              row, col = divmod(pos, 8)
319              sq = chess.square(col, row)
320              if state.piece_at(sq) is not None:
321                  continue
322              state.set_piece_at(sq, chess.Piece.from_symbol(piece))
323              break
324          return state

338          with open(self.epdeval, mode="r") as f:

350      def fen_random_generator(self, get_score, _limit):
351          board = chess.Board()
352          count = 0
353          endgames = []
354          pieces = ['P', 'p', 'N', 'n', 'B', 'b', 'R', 'r', 'Q', 'q']
355          for _ in range(_limit):
356              board.clear()
357              for king in ['K', 'k']:
358                  board = self.set_piece(state=board, piece=king)
359              c = self.random.choice([1, 2, 3, 4])
360              for _ in range(c):
361                  piece = self.random.choice(pieces)
362                  board = self.set_piece(state=board, piece=piece)
363              board.turn = chess.WHITE
364              fen_positions = board.fen()
365              if board.is_valid() and fen_positions not in endgames:
366                  if get_score(fen_positions) is not None:
367                      count += 1
368                      endgames.append(fen_positions)
369              board.turn = chess.BLACK
370              fen_positions = board.fen()
371              if board.is_valid() and fen_positions not in endgames:
372                  if get_score(fen_positions) is not None:
373                      count += 1
374                      endgames.append(fen_positions)
375              board.turn = chess.WHITE
376              for s in [chess.A6, chess.B6, chess.C6, chess.D6,
377                        chess.E6, chess.F6, chess.G6, chess.H6]:
378                  board.ep_square = s
379                  fen_positions = board.fen()
380                  if board.is_valid() and fen_positions not in endgames:
381                      if get_score(fen_positions) is not None:
382                          count += 1
383                          endgames.append(fen_positions)
384              board.turn = chess.BLACK
385              for s in [chess.A3, chess.B3, chess.C3, chess.D3,
386                        chess.E3, chess.F3, chess.G3, chess.H3]:
387                  board.ep_square = s
388                  fen_positions = board.fen()
389                  if board.is_valid() and fen_positions not in endgames:
390                      if get_score(fen_positions) is not None:
391                          count += 1
392                          endgames.append(fen_positions)
393              utils_progress(f"{count}/{limit} | {fen_positions}")
394          return endgames
